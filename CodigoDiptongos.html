        function crearTarjeta(id, esDiptongo = false) {

            const info = esDiptongo ? DATA_DIPTONGOS[id] : DATA_LETRAS[id];

            if (!info) return "";

            const principal = esDiptongo ? (modoMayuscula ? id.toUpperCase() : id) : (modoMayuscula ? id : info.m);

            const apoyo = esDiptongo ? "" : (modoMayuscula ? info.m : id);

            const pathLetras = esDiptongo ? (modoMayuscula ? info.pathMay : info.pathMin) : (modoMayuscula ? info.pathMay : info.pathMin);

            const puente = esDiptongo ? '<path class="puente-rojo" d="M 10,92 Q 60,102 110,92" />' : '';

            return `<div class="letter-card ${esDiptongo ? 'diptongo-card' : ''}" onclick="enseñarLetra('${id}', ${esDiptongo})">

                    <div class="mini-label">${apoyo}</div>

                    <div class="illustration">${info.e || ""}</div>

                    <svg viewBox="0 0 120 100">${RENGLONES_SVG}${puente}${pathLetras}</svg>

                    <div class="label">${principal}</div>

                </div>`;

        }

        function toggleModo() {

            modoMayuscula = !modoMayuscula;

            document.getElementById('btn-switch').innerText = modoMayuscula ? "ABC" : "abc";

            document.getElementById('btn-switch').classList.toggle('active', modoMayuscula);

            inicializarApp();

            cambiarSeccion(seccionActual);

        }

        function cambiarSeccion(id) {

            seccionActual = id;

            if (audioActual) { audioActual.pause(); audioActual = null; }

            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));

            document.querySelectorAll('.sections-nav .nav-btn').forEach(b => b.classList.remove('active'));

            document.getElementById(id).classList.add('active');

            if(document.getElementById('btn-' + id)) document.getElementById('btn-' + id).classList.add('active');

        }

        function enseñarLetra(id, esDiptongo) {

            const card = event.currentTarget;

            card.classList.remove('animating');

            void card.offsetWidth;

            card.classList.add('animating');

            if (audioActual) { audioActual.pause(); audioActual.currentTime = 0; }

            const info = esDiptongo ? DATA_DIPTONGOS[id] : DATA_LETRAS[id];

            if (info && info.url) { audioActual = new Audio(info.url); audioActual.play(); }

        }

        }
